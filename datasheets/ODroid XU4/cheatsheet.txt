# generate wpa configuration
>wpa_passphrase "lorem ipsum dolor sit amet" 7386801780590940 >> /etc/wpa_supplicant.conf
>wpa_passphrase -B "lorem ipsum dolor sit amet" 7386801780590940 >> /etc/wpa_supplicant.conf

# add the following params to etc/wpa_supplicant.conf
network={
        ssid="lorem ipsum dolor sit amet"
        psk="7386801780590940"
        proto=RSN
        key_mgmt=WPA-PSK
        pairwise=CCMP TKIP
        group=CCMP TKIP
        #psk=8df9408615472876e09ba00a81fccefcec75f8c497ca54cd8f5d9d3fa7a791ae
}

# start network 
>sudo wpa_supplicant -Dwext -iwlan0 -c/etc/wpa_supplicant.conf
# start network as deamon

>sudo wpa_supplicant -B -Dwext -iwlan0 -c/etc/wpa_supplicant.conf
